**Onde:** `repo-launcher.sh:388-415` (substituir corpo da funĂ§ĂŁo).

**Novo formato de saĂ­da:** TAB-separated, 6 colunas:

```
<type>  <name>  <head_marker>  <track_info>  <relative_time>  <merged_flag>
```

- `type`: `current` | `local` | `remote`
- `name`: refname:short (sem `refs/heads/` ou `refs/remotes/`)
- `head_marker`: `*` se HEAD, vazio se nĂŁo
- `track_info`: output do `%(upstream:track,nobracket)` â€” pode ser `ahead 3, behind 1`, `gone`, ou vazio
- `relative_time`: `2 hours ago`, `3 days ago`, etc. (`%(committerdate:relative)`)
- `merged_flag`: `merged` se branch local estĂˇ em `git branch --merged <default>`, senĂŁo vazio

**ImplementaĂ§ĂŁo principal:**

```bash
list_branches() {
  local repo_path="$1"
  (
    cd "$repo_path" || exit 1

    # Default branch detection (para merged check)
    local default_branch
    default_branch=$(git symbolic-ref --short refs/remotes/origin/HEAD 2>/dev/null | sed 's|^origin/||')
    if [[ -z "$default_branch" ]]; then
      # Fallback: tentar main, master, ou primeira local
      for candidate in main master; do
        if git rev-parse --verify "$candidate" >/dev/null 2>&1; then
          default_branch="$candidate"
          break
        fi
      done
    fi

    # Set de branches mergeadas (vazio se nĂŁo conseguir detectar default)
    local merged_set=""
    if [[ -n "$default_branch" ]]; then
      merged_set=$(git branch --merged "$default_branch" --format='%(refname:short)' 2>/dev/null | tr '\n' '|')
      merged_set="|${merged_set}|"
    fi

    local current
    current=$(git branch --show-current 2>/dev/null)

    # Locais
    git for-each-ref \
      --format='%(refname:short)|%(upstream:track,nobracket)|%(committerdate:relative)' \
      refs/heads/ 2>/dev/null | while IFS='|' read -r name track when; do
      [[ -z "$name" ]] && continue
      local merged=""
      [[ -n "$merged_set" && "$merged_set" == *"|${name}|"* && "$name" != "$default_branch" ]] && merged="merged"
      local btype="local"
      [[ "$name" == "$current" ]] && btype="current"
      printf '%s\t%s\t\t%s\t%s\t%s\n' "$btype" "$name" "$track" "$when" "$merged"
    done

    # Remotas (filtrar HEAD e branches que jĂˇ tĂŞm cĂłpia local)
    local local_branches
    local_branches="|$(git branch --format='%(refname:short)' 2>/dev/null | tr '\n' '|')|"
    git for-each-ref \
      --format='%(refname:short)|%(committerdate:relative)' \
      refs/remotes/ 2>/dev/null | while IFS='|' read -r name when; do
      [[ -z "$name" ]] && continue
      [[ "$name" != */* ]] && continue
      [[ "$name" == */HEAD ]] && continue
      local bare="${name#*/}"
      [[ "$local_branches" == *"|$bare|"* ]] && continue
      printf 'remote\t%s\t\t\t%s\t\n' "$name" "$when"
    done
  )
}
```

**Notas:**
- Tab-separated, 6 colunas (head_marker e merged_flag entre os outros). Tabs vazios para colunas inexistentes.
- Locals fazem 2 calls extra (`branch --show-current`, `branch --merged`) â€” aceitĂˇvel.
- Remote skip `*/HEAD` e dedup com locais â€” comportamento original preservado.
- Erro silencioso (`2>/dev/null`) â€” funĂ§ĂŁo robusta a repos quebrados.

abbrev_time() {
  local input="$1"
  # "2 hours ago" â†’ "2h"; "1 day ago" â†’ "1d"; "3 weeks ago" â†’ "3w"; "5 minutes ago" â†’ "5m"
  local n unit
  n=$(printf '%s' "$input" | awk '{print $1}')
  unit=$(printf '%s' "$input" | awk '{print $2}')
  case "$unit" in
    minute|minutes) printf '%sm' "$n" ;;
    hour|hours)     printf '%sh' "$n" ;;
    day|days)       printf '%sd' "$n" ;;
    week|weeks)     printf '%sw' "$n" ;;
    month|months)   printf '%smo' "$n" ;;
    year|years)     printf '%sy' "$n" ;;
    *) printf '%s' "$input" ;;   # fallback (e.g., "just now")
  esac
}
